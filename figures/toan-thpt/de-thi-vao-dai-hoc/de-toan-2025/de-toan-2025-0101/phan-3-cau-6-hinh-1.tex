% Đặt kích thước chính xác
\def\a{10.4} % Cạnh đáy lớn
\def\b{7.4}  % Cạnh đáy nhỏ
\def\h{1.5}   % Chiều cao

% Đáy dưới ABCD (hình vuông)

\coordinate (B) at (\a,0,0);
\coordinate (C) at (\a,\a,0);
\coordinate (D) at (0,\a,0);

\coordinate (Ap) at ({(\a-\b)/2},{(\a-\b)/2},\h);
\coordinate (Bp) at ({(\a+\b)/2},{(\a-\b)/2},\h);
\coordinate (Cp) at ({(\a+\b)/2},{(\a+\b)/2},\h);
\coordinate (Dp) at ({(\a-\b)/2},{(\a+\b)/2},\h);

% Vẽ đáy dưới
\draw[dashed] (A) -- (B);
\draw[dashed] (A) -- (D);
\draw (B) -- (C);
\draw (C) -- (D);

% Vẽ đáy trên
\draw (Ap) -- (Bp) -- (Cp) -- (Dp) -- cycle;

% Vẽ các cạnh bên
\draw[dashed] (A) -- (Ap);
\draw (B) -- (Bp);
\draw (C) -- (Cp);
\draw (D) -- (Dp);

% Ghi chú kích thước (tùy chọn)

\draw ($(C)+(1.5,0,0)$) -- (C);
\draw ($(B)+(1.5,0,0)$) -- (B);

\draw[<->, dashed] ($(Ap)+(0,0,.5)$) -- ($(Dp)+(0,0,.5)$) node[midway, above] {\small 7,4 cm};
